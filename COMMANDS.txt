       `id`    INTEGER PRIMARY KEY AUTOINCREMENT,
       `email` TEXT NOT NULL UNIQUE,
       `firstname`     TEXT NOT NULL,
       `lastname`      TEXT NOT NULL,
       `username`      TEXT NOT NULL UNIQUE,
       `password`      TEXT NOT NULL,
       `profile_pic`   TEXT DEFAULT “default.jpg”,
       `profile_bio`   TEXT,
       `created_at`    TEXT,
);
